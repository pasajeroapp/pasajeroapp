L.control.locate({ position: 'topleft', strings: { title: "Mostrar mi ubicación" }, flyTo: true }).addTo(map); // Activar geolocalización map.locate({setView: true, maxZoom: 16}); // Mostrar marcador en la ubicación actual function onLocationFound(e) { var radius = e.accuracy / 2; L.marker(e.latlng).addTo(map) .bindPopup("Estás aquí").openPopup(); L.circle(e.latlng, radius, {color: 'blue', fillOpacity: 0.2}).addTo(map); } map.on('locationfound', onLocationFound); // Manejar errores de geolocalización function onLocationError(e) { alert(e.message); } map.on('locationerror', onLocationError);
